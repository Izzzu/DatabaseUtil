CREATE table IF NOT EXISTS situations  (
id_sit INT PRIMARY KEY NOT NULL GENERATED BY DEFAULT AS IDENTITY(START WITH 0),
term VARCHAR(30) NOT NULL,
LCU INT )

CREATE table IF NOT EXISTS chatbotAnswers (
id_ans INT PRIMARY KEY NOT NULL GENERATED BY DEFAULT AS IDENTITY(START WITH 0),
answer VARCHAR(100) NOT NULL,
id_sit INT,
chatLevel INT,
prevStatNote INT,
expStatTable VARCHAR(10)
}
  
CREATE table IF NOT EXISTS adverb (
id_adv INT PRIMARY KEY NOT NULL GENERATED BY DEFAULT AS IDENTITY(START WITH 0),
adverb VARCHAR(100) NOT NULL,
statNote INT
}
